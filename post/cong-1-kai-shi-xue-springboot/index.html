use pms;#新建的数据库名
SET FOREIGN_KEY_CHECKS=0;
DROP TABLE IF EXISTS `u_user`;
CREATE TABLE `u_user` (
  `u_id` bigint NOT NULL AUTO_INCREMENT COMMENT 'id标识符 自增长',
  `u_name` varchar(255) NOT NULL COMMENT '用户姓名',
  `u_sex` tinyint NOT NULL COMMENT '用户性别 1、男性 2、女性 ',
  `u_birthday` date NOT NULL COMMENT '用户年龄 出生年月日',
  `u_state` tinyint NOT NULL DEFAULT '1' COMMENT '用户状态 1、正常 2、删除',
  `u_ascription` tinyint DEFAULT NULL COMMENT '归属',
  PRIMARY KEY (`u_id`)
) ENGINE=InnoDB AUTO_INCREMENT=120 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
