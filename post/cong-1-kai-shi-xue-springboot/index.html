完成后就开始写实体类 User

@Data//lombok
@JsonInclude(JsonInclude.Include.NON_NULL)//json值为null时不传递
public class User {
    private long id;//id
    private String name;//姓名
    private byte sex;//性别
    private String birthday;//生日
    private byte state;//状态
    private long ascription;//从属
}

实体类就照着数据库建就好了 id要用long 因为int型的取值范围是-2147483648~2147483647，在MySQL里的id字段通常设置成long类型而不是int型，是为了后续的使用兼容，随着业务增长，大数据的部署，还有第三方的接入使用，比如微信支付宝，它们返回的值。 用它就对了

其中性别状态从属都用了byte 因为性别、状态可能会有很多种类型 像性别1表示男性，2表示女性，3表示未知，而byte是最小的数据类型 byte的取值范围是0-255 选它就完事了
接下来是 UserService

public interface UserService {
        /**
         * 增加
         */
        int insertU(User user);

        /**
         * 删除
         */
        int deleteU(long[] ids);

        /**
         * 更改
         */
        int updateU(User user);

        /**
         * 根据id查找
         */
        List<User> getUById(long[] ids);

        /**
         * 查找全部
         */
        List<User> getAll(User user);
}

UserServiceImpl

@Service
public class UserServiceImpl implements UserService {
    @Autowired
    private UserMapper userMapper;
    @Override
    public int insertU(User user) {
        return userMapper.insertU(user);
    }

    @Override
    public int deleteU(long[] ids) {
        return userMapper.deleteU(ids);
    }

    @Override
    public int updateU(User user) {
        return userMapper.updateU(user);
    }

    @Override
    public List<User> getUById(long[] ids) {
        return userMapper.getUById(ids);
    }

    @Override
    public List<User> getAll(User user) {
        return userMapper.getAll(user);
    }
}

UserController

@RestController
@RequestMapping("/user")
public class UserController {
    // http://localhost:8080/user/
    @Autowired
    private UserService userService;

    /**
     *
     * @param user 插入数据
     * @return 影响行数
     */
    @RequestMapping(value = "/insertU")
    public  int insertU(User user){
        return userService.insertU(user);
    }

    /**
     *
     * @param ids 根据id删除
     * @return  影响行数
     */
    @RequestMapping(value = "/deleteU")
    public int deleteU(long[] ids) {
        return userService.deleteU(ids);
    }

    /**
     *
     * @param user 更新用户
     * @return  影响行数
     */
    @RequestMapping(value = "/updateU")
    public int updateU(User user) {
        return userService.updateU(user);
    }

    /**
     *
     * @param ids 根据id查找
     * @return 查找内容
     */
    @RequestMapping(value = "/getUById")
    public List<User> getUById(long[] ids) {
        return userService.getUById(ids);
    }

    /**
     *
     * @param user 查找所有用户
     * @return 查找内容
     */
    @RequestMapping(value = "/getAll")
    public List<User> getAll(User user){
        return userService.getAll(user);
    }
}

UserMapper.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.example.pms.mapper.user.UserMapper">

    <insert id="insertU" >
        insert into pms.u_user(u_name, u_sex, u_birthday, u_state, u_ascription)
        VALUES (#{name},#{sex},#{birthday},#{state},#{ascription})
    </insert>

    <delete id="deleteU" >
        update pms.u_user
        set u_state=2
        where u_id in
        <foreach collection="ids" open="(" separator="," close=")" item="id">
            #{id}
        </foreach>
    </delete>

    <update id="updateU" >
        update pms.u_user
        set u_name=#{name}, u_sex=#{sex}, u_birthday=#{birthday},u_state=#{state}, u_ascription=#{ascription}
        where u_id=#{id}
    </update>

    <select id="getUById" resultType="com.example.pms.entity.user.User">
        select  u_id id,u_name name, u_sex sex, u_birthday birthday, u_state state, u_ascription ascription
        from pms.u_user
        where u_id in
        <foreach collection="ids" open="(" separator="," close=")" item="id">
            #{id}
        </foreach>

    </select>

    <select id="getAll" resultType="com.example.pms.entity.user.User" >
        select u_id id, u_name name, u_sex sex, u_birthday birthday, u_state state, u_ascription ascription
        from pms.u_user
    </select>

</mapper>
